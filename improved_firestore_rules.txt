rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // User profiles - users can only read/write their own profile
    match /users/{userId} {
      allow read: if request.auth != null && request.auth.uid == userId;
      allow write: if request.auth != null && request.auth.uid == userId;
      
      // Optionally allow other authenticated users to read basic profile info
      // for features like user search, connections, etc.
      allow read: if request.auth != null && 
                    request.resource.data.keys().hasOnly(['uid', 'username', 'connectionStatus']);
    }
    
    // Messages - users can only access conversations they're part of
    match /messages/{conversationId} {
      allow read, write: if request.auth != null && 
                           conversationId.matches(request.auth.uid + '_.*') || 
                           conversationId.matches('.*_' + request.auth.uid);
    }
    
    // Default deny
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
